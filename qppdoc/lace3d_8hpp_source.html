738  template<class VALTYPE>
739  vector3d<typename numeric_type<VALTYPE>::complex> solve_cubeq(VALTYPE a, VALTYPE b, VALTYPE c, VALTYPE d)
740  // Solves ax^3 + bx^2 + cx + d = 0
741  {
742  b /= a;
743  c /= a;
744  d /= a;
745  typename numeric_type<VALTYPE>::complex I(VALTYPE(0),VALTYPE(1));
746  VALTYPE aa = c/3 - b*b/9, bb = d/2 - b*c/6 + b*b*b/27;
747  VALTYPE discr1 = aa*aa*aa + bb*bb;
748  typename numeric_type<VALTYPE>::complex discr2;
749  if (discr1 > VALTYPE(0))
750  {
751  VALTYPE discr23 = std::sqrt(discr1) - bb;
752  int sgn = (discr23>0)? 1 : -1;
753  discr23 *= sgn;
754  discr2 = sgn*std::exp(std::log(discr23)/3);
755  }
756  else
757  {
758  discr2 = -bb + std::sqrt(-discr1)*I;
759  VALTYPE dabs = std::abs(discr2), darg = std::arg(discr2);
760  discr2 = std::polar(std::exp(std::log(dabs)/3),darg/3);
761  }
762  typename numeric_type<VALTYPE>::complex r1 = discr2 - aa/discr2, r2 = discr2 + aa/discr2;
763  return vector3d<typename numeric_type<VALTYPE>::complex>( r1 - b/3,
764  -r1/2 + VALTYPE(std::sqrt(3.0))*r2*I/2 - b/3,
765  -r1/2 - VALTYPE(std::sqrt(3.0))*r2*I/2 - b/3);
766  }
